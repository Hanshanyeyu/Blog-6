# 大数相加（极简版）



```js

/** 
 * 2020.02.27
 * @param {String} a 
 * @param {String} b
 * @return {String} 
 */

function add_test(a, b) {
	// init
	a = a.split('')
	b = b.split('')
	let result = ''
	let temp = 0
	let carry = 0  //是否进位

	// iterative part
	while (a.length || b.length || carry) {
		temp = parseInt(a.pop() || 0) + parseInt(b.pop() || 0) + carry
		result = temp % 10 + result
		carry = Math.floor(temp / 10)
	}

	// return 
	return result
}

console.log(add_test('2222222222222222', '33333333333333333'));
```
